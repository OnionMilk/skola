    $ s = (2*k + 1) * \frac{\lambda}{4}\ \ \  $ (grundton: $ k = 0 $)

    $ s = 10\ cm = 0.1\ m $

    $ \lambda = \frac{v}{f} $

    $ s = \frac{\frac{v}{f}}{4} \rightarrow\ f = \frac{v}{4*s} $

    $ v = 350\ m/s $

    $ f = \frac{350}{4*0.1} = 875\ Hz $

    \hfill

\item[\textbf{iii.}]
    $ f = \frac{875}{2} = 437.5\ Hz $

    \hfill

\end{itemize}
